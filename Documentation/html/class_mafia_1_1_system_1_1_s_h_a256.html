59 {
60  // Appends a 1 bit to the input, then pads with 0 bits
61  // until the length of the input is congruent to 448 modulo 512.
62  // Finally appends the size in bits of the input as it was before the
63  // padding, as a 64-bit big-endian integer.
64  // If the system the algorithm is executed on is not big-endian, the
65  // longword inputSize is converted to big-endian.
66  // Returns the result of the preprocessing.
67

69  EXTENDED_PADDING = 0x00;
70

72 
73  input.insert(input.end(), 1, INITIAL_PADDING);
74  for(int i = input.size() * BYTE_SIZE; (i % CHUNK_SIZE) != SIZE_WITHOUT_LENGTH; i += BYTE_SIZE)
75  {
76  input.insert(input.end(), 1, EXTENDED_PADDING); // Build byte by byte and keep checking the size.
77  }

109  }
